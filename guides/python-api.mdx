## Quick Start

```python
from acestep.handler import AceStepHandler
from acestep.llm_inference import LLMHandler
from acestep.inference import GenerationParams, GenerationConfig, generate_music

# Initialize handlers
dit_handler = AceStepHandler()
llm_handler = LLMHandler()

# Initialize services
dit_handler.initialize_service(
    project_root="/path/to/ACE-Step-1.5",
    config_path="acestep-v15-turbo",
    device="cuda"
)

llm_handler.initialize(
    checkpoint_dir="/path/to/checkpoints",
    lm_model_path="acestep-5Hz-lm-0.6B",
    backend="vllm",
    device="cuda"
)

# Configure generation
params = GenerationParams(
    caption="upbeat electronic dance music with heavy bass",
    bpm=128,
    duration=30,
)

config = GenerationConfig(
    batch_size=2,
    audio_format="flac",
)

# Generate music
result = generate_music(
    dit_handler,
    llm_handler,
    params,
    config,
    save_dir="output"
)

# Access results
if result.success:
    for audio in result.audios:
        print(f"Generated: {audio['path']}")
        print(f"Seed: {audio['params']['seed']}")
else:
    print(f"Error: {result.error}")
```

## Core Components

### Handler Classes

#### AceStepHandler (DiT Model)

Manages the Diffusion Transformer model:

```python
from acestep.handler import AceStepHandler

dit_handler = AceStepHandler()

dit_handler.initialize_service(
    project_root="/path/to/ACE-Step-1.5",
    config_path="acestep-v15-turbo",  # or "acestep-v15-base"
    device="cuda",
    use_flash_attention=True,
    offload_to_cpu=False,
    offload_dit_to_cpu=False,
    quantization=None,  # or "int8_weight_only"
    compile_model=True
)
```

**Parameters:**

<ParamField path="project_root" type="string" required>
  Path to ACE-Step project directory
</ParamField>

<ParamField path="config_path" type="string" default="acestep-v15-turbo">
  DiT model configuration name or path
</ParamField>

<ParamField path="device" type="string" default="auto">
  Device: `auto`, `cuda`, `cpu`, `mps`, `xpu`
</ParamField>

<ParamField path="use_flash_attention" type="boolean" default="True">
  Enable flash attention for faster inference
</ParamField>

<ParamField path="offload_to_cpu" type="boolean" default="False">
  Offload models to CPU when idle (auto-enabled on GPUs &lt;20GB)
</ParamField>

<ParamField path="quantization" type="string" default="None">
  Quantization mode: `None` or `"int8_weight_only"`
</ParamField>

#### LLMHandler (5Hz Language Model)

Manages the 5Hz semantic language model:

```python
from acestep.llm_inference import LLMHandler

llm_handler = LLMHandler()

llm_handler.initialize(
    checkpoint_dir="/path/to/checkpoints",
    lm_model_path="acestep-5Hz-lm-0.6B",  # or 1.7B, 4B
    backend="vllm",  # or "pt", "mlx"
    device="cuda",
    offload_to_cpu=False
)
```

**Backends:**
- `vllm` - Faster, NVIDIA GPUs ≥8GB VRAM (recommended)
- `pt` - PyTorch, universal fallback
- `mlx` - Apple Silicon optimized

### GenerationParams

Defines all music generation parameters:

```python
from acestep.inference import GenerationParams

params = GenerationParams(
    # Task & Instruction
    task_type="text2music",  # or "cover", "repaint", etc.
    instruction="Fill the audio semantic mask...",
    
    # Text Inputs
    caption="upbeat pop rock with electric guitars",
    lyrics="""[Verse 1]
    Walking down the street today
    
    [Chorus]
    I'm moving on, I'm staying strong
    """,
    instrumental=False,
    
    # Metadata
    vocal_language="en",
    bpm=120,
    keyscale="C Major",
    timesignature="4",
    duration=45.0,
    
    # Audio Inputs (optional)
    reference_audio="path/to/reference.mp3",
    src_audio="path/to/source.mp3",
    audio_codes="",  # Pre-computed codes
    
    # DiT Parameters
    inference_steps=8,
    guidance_scale=7.0,
    seed=-1,
    shift=3.0,  # Timestep shift (recommended for turbo)
    infer_method="ode",  # or "sde"
    timesteps=None,  # Custom schedule
    
    # Advanced DiT
    use_adg=False,
    cfg_interval_start=0.0,
    cfg_interval_end=1.0,
    
    # Task-Specific
    repainting_start=0.0,
    repainting_end=-1,
    audio_cover_strength=1.0,
    
    # LM Parameters
    thinking=True,
    lm_temperature=0.85,
    lm_cfg_scale=2.0,
    lm_top_k=0,
    lm_top_p=0.9,
    lm_negative_prompt="NO USER INPUT",
    
    # CoT Options
    use_cot_metas=True,
    use_cot_caption=True,
    use_cot_lyrics=False,
    use_cot_language=True,
    use_constrained_decoding=True,
)
```

### GenerationConfig

Controls batch processing and output:

```python
from acestep.inference import GenerationConfig

config = GenerationConfig(
    batch_size=2,  # Number of variations
    allow_lm_batch=True,  # Batch LM operations
    use_random_seed=True,  # Random seeds per batch
    seeds=None,  # or [42, 123, 456]
    lm_batch_chunk_size=8,  # Max LM batch size
    constrained_decoding_debug=False,
    audio_format="flac",  # "mp3", "wav", "flac"
)
```

### GenerationResult

Returned by `generate_music()`:

```python
@dataclass
class GenerationResult:
    # Audio Outputs
    audios: List[Dict[str, Any]]
    # Each audio dict:
    # {
    #   "path": str,           # File path
    #   "tensor": Tensor,      # Audio data [channels, samples]
    #   "key": str,            # Unique ID
    #   "sample_rate": int,    # 48000
    #   "params": Dict,        # Generation params (seed, codes, etc.)
    # }
    
    # Generation Info
    status_message: str
    extra_outputs: Dict[str, Any]
    # Contains:
    # - "lm_metadata": LM-generated metadata
    # - "time_costs": Timing information
    # - "latents": Intermediate tensors
    
    # Status
    success: bool
    error: Optional[str]
```

## Main Functions

### generate_music()

Primary music generation function:

```python
def generate_music(
    dit_handler: AceStepHandler,
    llm_handler: LLMHandler,
    params: GenerationParams,
    config: GenerationConfig,
    save_dir: Optional[str] = None,
    progress=None,
) -> GenerationResult
```

**Example:**

```python
result = generate_music(
    dit_handler,
    llm_handler,
    params,
    config,
    save_dir="output"
)

if result.success:
    print(f"Generated {len(result.audios)} audio files")
    for audio in result.audios:
        print(f"  {audio['path']} (seed: {audio['params']['seed']})")
    
    # Access time costs
    time_costs = result.extra_outputs.get("time_costs", {})
    print(f"DiT time: {time_costs.get('dit_total_time_cost', 0):.2f}s")
    print(f"Total time: {time_costs.get('pipeline_total_time', 0):.2f}s")
else:
    print(f"Generation failed: {result.error}")
```

### create_sample()

Generate complete sample from description (Simple Mode):

```python
def create_sample(
    llm_handler: LLMHandler,
    query: str,
    instrumental: bool = False,
    vocal_language: Optional[str] = None,
    temperature: float = 0.85,
    top_k: Optional[int] = None,
    top_p: Optional[float] = None,
    repetition_penalty: float = 1.0,
    use_constrained_decoding: bool = True,
    constrained_decoding_debug: bool = False,
) -> CreateSampleResult
```

**Example:**

```python
from acestep.inference import create_sample

result = create_sample(
    llm_handler=llm_handler,
    query="energetic K-pop dance track with catchy hooks",
    vocal_language="ko",
    temperature=0.85
)

if result.success:
    # Use generated metadata for music generation
    params = GenerationParams(
        caption=result.caption,
        lyrics=result.lyrics,
        bpm=result.bpm,
        duration=result.duration,
        keyscale=result.keyscale,
        vocal_language=result.language,
        thinking=True
    )
    
    music_result = generate_music(dit_handler, llm_handler, params, config)
```

### format_sample()

Enhance user-provided caption and lyrics:

```python
def format_sample(
    llm_handler: LLMHandler,
    caption: str,
    lyrics: str,
    user_metadata: Optional[Dict[str, Any]] = None,
    temperature: float = 0.85,
    top_k: Optional[int] = None,
    top_p: Optional[float] = None,
    repetition_penalty: float = 1.0,
    use_constrained_decoding: bool = True,
    constrained_decoding_debug: bool = False,
) -> FormatSampleResult
```

**Example:**

```python
from acestep.inference import format_sample

result = format_sample(
    llm_handler=llm_handler,
    caption="rock ballad",
    lyrics="[Verse]\nIn the darkness I find my way...",
    user_metadata={"bpm": 72},  # Constrain specific values
    temperature=0.85
)

if result.success:
    params = GenerationParams(
        caption=result.caption,  # Enhanced
        lyrics=result.lyrics,    # Formatted
        bpm=result.bpm,
        keyscale=result.keyscale,
        thinking=True,
        use_cot_metas=False  # Already formatted
    )
```

### understand_music()

Analyze audio codes to extract metadata:

```python
def understand_music(
    llm_handler: LLMHandler,
    audio_codes: str,
    temperature: float = 0.85,
    top_k: Optional[int] = None,
    top_p: Optional[float] = None,
    repetition_penalty: float = 1.0,
    use_constrained_decoding: bool = True,
    constrained_decoding_debug: bool = False,
) -> UnderstandResult
```

**Example:**

```python
from acestep.inference import understand_music

# Analyze codes from previous generation
audio_codes = result.audios[0]['params']['audio_codes']

analysis = understand_music(
    llm_handler=llm_handler,
    audio_codes=audio_codes,
    temperature=0.85
)

if analysis.success:
    print(f"Caption: {analysis.caption}")
    print(f"BPM: {analysis.bpm}")
    print(f"Key: {analysis.keyscale}")
    print(f"Language: {analysis.language}")
```

## Complete Examples

### Example 1: Basic Text-to-Music

```python
from acestep.handler import AceStepHandler
from acestep.llm_inference import LLMHandler
from acestep.inference import GenerationParams, GenerationConfig, generate_music

# Initialize
dit_handler = AceStepHandler()
llm_handler = LLMHandler()

dit_handler.initialize_service(
    project_root=".",
    config_path="acestep-v15-turbo",
    device="cuda"
)

llm_handler.initialize(
    checkpoint_dir="./checkpoints",
    lm_model_path="acestep-5Hz-lm-0.6B",
    backend="vllm",
    device="cuda"
)

# Generate
params = GenerationParams(
    caption="calm ambient music with soft piano and strings",
    duration=60,
    bpm=80,
    keyscale="C Major",
    thinking=True
)

config = GenerationConfig(
    batch_size=2,
    audio_format="flac"
)

result = generate_music(dit_handler, llm_handler, params, config, save_dir="output")

for i, audio in enumerate(result.audios, 1):
    print(f"Output {i}: {audio['path']}")
```

### Example 2: Batch Processing

```python
import json
from pathlib import Path

output_dir = Path("batch_output")
output_dir.mkdir(exist_ok=True)

prompts = [
    "upbeat pop rock",
    "melancholic jazz",
    "energetic EDM",
    "classical piano"
]

for i, prompt in enumerate(prompts):
    print(f"\nGenerating {i+1}/{len(prompts)}: {prompt}")
    
    params = GenerationParams(
        caption=prompt,
        lyrics="[Instrumental]",
        instrumental=True,
        duration=30,
        thinking=True
    )
    
    config = GenerationConfig(
        batch_size=1,
        use_random_seed=True,
        audio_format="mp3"
    )
    
    result = generate_music(
        dit_handler,
        llm_handler,
        params,
        config,
        save_dir=str(output_dir / f"batch_{i}")
    )
    
    if result.success:
        # Save metadata
        metadata = {
            "prompt": prompt,
            "file": result.audios[0]['path'],
            "seed": result.audios[0]['params']['seed'],
            "bpm": result.extra_outputs.get('lm_metadata', {}).get('bpm'),
            "key": result.extra_outputs.get('lm_metadata', {}).get('keyscale')
        }
        
        with open(output_dir / f"batch_{i}_meta.json", "w") as f:
            json.dump(metadata, f, indent=2)
        
        print(f"  ✓ Saved: {result.audios[0]['path']}")
    else:
        print(f"  ✗ Failed: {result.error}")

print(f"\nBatch complete! Output in {output_dir}")
```

## Memory Management

ACE-Step 1.5 includes automatic VRAM management:

- **Automatic tier detection** - Detects GPU VRAM and selects optimal settings
- **VRAM guard** - Estimates requirements and reduces batch size if needed
- **Adaptive VAE decode** - Three-tier fallback (GPU tiled → GPU with CPU offload → full CPU)
- **Auto chunk sizing** - Adapts to free VRAM (64/128/256/512/1024/1536)

Manual tuning if needed:

```python
dit_handler.initialize_service(
    project_root=".",
    config_path="acestep-v15-turbo",
    device="cuda",
    offload_to_cpu=True,  # Enable for &lt;20GB VRAM
    quantization="int8_weight_only"  # Enable for &lt;20GB VRAM
)

config = GenerationConfig(
    batch_size=1,  # Reduce if OOM
    lm_batch_chunk_size=2  # Reduce for LM on low VRAM
)
```
